// Arrow Function
function add(a,b) {
	console.log(a+b);
}

add = (a,b) => {console.log(a+b)}
add(7,3); // 10

// Pure Function

function test(a){
	console.log(a*5);
}
test(10); //


for(var i = 0; i<5 ;i++) {
	setTimeout(()=>console.log(i));
}
// 5 times 5 due to call stack, event loop, callback queue

for(let i = 0; i<5 ;i++) {
	setTimeout(()=>console.log(i));
}
// 0, 1, 2, 3, 4 due to scope of let
